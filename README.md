# my-project

git init
- iniciar um novo projeto com git

git add <nome-arquivo> ou .
- adiciona os arquivos que estão prontos para serem commitados

git commit -m "minha-mensagem"
- commita os arquivos no historico

git log
- mostra o historico de commits

git statuts
- mostra como esta o estado da nossa ramificação(branch)

git diff
- mostra as alterações da ramificação atual

git merge 
- mescla ramificações

git branch
- mostra a branch atual

git branch <nome-branch>
- cria uma nova branch a partir do historico da branch atual

git checkout <nome-branch>
- muda para essa branch

git remote add <nome> <url>
- adiciona um novo repositório remoto

git push <nome> <nome-branch>
- manda as alterações locais para o repositorio remoto

git pull <nome> <nome-branch> 
- pega as alterações do repositório remoto e traz elas para o local

git fetch
- atualiza o novo historico local de acordo com o historico salvo do repositorio remoto
- sincronização do local com o remoto
